- Comenzando con este código, ordena el []int y []string.

    package main

    import (
        "fmt"
    )

    func main() {
        xi := []int{5, 8, 2, 43, 17, 987, 14, 12, 21, 1, 4, 2, 3, 93, 13}
        xs := []string{"Una", "persona", "que", "nunca", "ha", "cometido", "un", "error", "nunca", "intenta", "nada", "nuevo"}

        fmt.Println(xi)
        // sort xi
        fmt.Println(xi)

        fmt.Println(xs)
        // sort xs
        fmt.Println(xs)
    }
